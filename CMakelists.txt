cmake_minimum_required(VERSION 3.15)
project(ytf_project LANGUAGES C CXX)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_C_STANDARD 11)

# ---------------------------------------------------------
# 1. Global Definitions
# ---------------------------------------------------------
add_definitions(-DUTF8PROC_STATIC)

if(MSVC)
    add_definitions(-D_CRT_SECURE_NO_WARNINGS -Dssize_t=SSIZE_T)
    # Dinamyc runtime /Md
    set(CMAKE_MSVC_RUNTIME_LIBRARY "MultiThreadedDLL")
endif()

# ---------------------------------------------------------
# 2. Fetch CURL 
# ---------------------------------------------------------
include(FetchContent)

FetchContent_Declare(
  curl
  GIT_REPOSITORY https://github.com/curl/curl.git
  GIT_TAG        curl-8_6_0
  GIT_SHALLOW    TRUE
)

# option for dynamic build
set(BUILD_SHARED_LIBS ON CACHE BOOL "" FORCE)
set(CURL_STATICLIB OFF CACHE BOOL "" FORCE)
set(BUILD_CURL_EXE OFF CACHE BOOL "" FORCE)
set(BUILD_TESTING OFF CACHE BOOL "" FORCE)

# Schannel for HTTPS on Windows
if(WIN32)
    set(CURL_USE_SCHANNEL ON CACHE BOOL "" FORCE)
endif()

FetchContent_MakeAvailable(curl)

# ---------------------------------------------------------
# 3. Include directories
# ---------------------------------------------------------
include_directories(include .)

# ---------------------------------------------------------
# 4. Sources
# ---------------------------------------------------------
set(MY_SOURCES
    src/ytf.cpp
    src/sqll.cpp
    src/types.cpp
    src/util.cpp
    src/console.cpp
)

set(DEP_SOURCES
    dep/sqlite3.c
    dep/utf8proc.c
)

# ---------------------------------------------------------
# 5. Executable
# ---------------------------------------------------------
add_executable(ytf ${MY_SOURCES} ${DEP_SOURCES})

# ---------------------------------------------------------
# 6. Linking
# ---------------------------------------------------------
# 
target_link_libraries(ytf PRIVATE libcurl)

if(WIN32)
    target_link_libraries(ytf PRIVATE
        Ws2_32
        Crypt32
        Normaliz
        advapi32
    )
else()
    target_link_libraries(ytf PRIVATE pthread dl m)
endif()
